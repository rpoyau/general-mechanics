\section{Tier‑1: Maxwell‑Style Field Analogue}
\label{sec:tier1-maxwell}

\subsection{T1M1 (Potential and curvature)}
Let $\mathcal A \in \Omega^{1}(\mathcal M)$ and define
\[
  F \;:=\; d\mathcal A \;\in\; \Omega^{2}(\mathcal M), \qquad dF=0.
\]

\subsection{T1M2 (Source equation)}
Fix a metric on $\mathcal M$ with Hodge map $\star$. Introduce $\Pi \in \Omega^{n-2}(\mathcal M)$ and set
\begin{equation}
  d\,\star F \;=\; J \;-\; \Pi, 
  \qquad
  d\Pi \;=\; \Sigma .
  \label{eq:maxwell-analogue}
\end{equation}

\subsection{T1M3 (Action)}
\begin{equation}
  \mathcal S[\mathcal A]
  \;=\;
  \frac{1}{2}\int_{\mathcal M} F \wedge \star F
  \;+\; \int_{\mathcal M} \mathcal A \wedge (J-\Pi) .
  \label{eq:maxwell-action}
\end{equation}
Stationarity yields \eqref{eq:maxwell-analogue}.
