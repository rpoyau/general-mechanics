\section{Continuity and world‑tube balance}\label{sec:worldtube}

\paragraph{Statement.}
Let $\Omega$ be a compact oriented $(n{+}1)$‑domain with boundary $\partial\Omega$ and inclusion
$\iota:\partial\Omega\hookrightarrow\Omega$. Assume D1 (\S\ref{sec:rim-dynamics}):
\begin{equation*}
\int_{\partial\Omega}\iota^{*}J \;=\; \int_{\Omega}\Sigma,\qquad \Sigma:=\mathrm dJ .
\end{equation*}
No metric is assumed.

\paragraph{Additivity.}
If $\Omega=\Omega_1\cup\Omega_2$ with interior interface $\Gamma$, opposite orientations on $\Gamma$
cancel, hence
\[
\int_{\partial\Omega}\iota^{*}J=
\int_{\partial\Omega_1}\iota^{*}J+\int_{\partial\Omega_2}\iota^{*}J,\quad
\int_{\Omega}\Sigma=
\int_{\Omega_1}\Sigma+\int_{\Omega_2}\Sigma .
\]

\paragraph{Boundary readouts.}
For measurable $W\subset\partial\Omega$, define $Q[W]:=\int_W \iota^{*}J$. Then $Q$ is finitely
additive on oriented decompositions of $\partial\Omega$.

\paragraph{Local form (choice of chart/volume).}
Given a chart $(t,x^i)$ and a reference volume, there exist component fields $(\rho,j^i,\sigma)$
such that
\[
\partial_t \rho + \partial_i j^i = \sigma ,
\]
a re‑expression of \eqref{eq:stokes-balance} in that description. No extra assumptions are added.

\paragraph{Small‑neighborhood limit.}
For a ball $B_\varepsilon(p)\subset\Omega$,
\[
\int_{\partial B_\varepsilon(p)}\!\iota^{*}J
= \int_{B_\varepsilon(p)}\!\Sigma
= \Sigma(p)\,\mathrm{Vol}(B_\varepsilon)+o(\varepsilon^{n+1}) \quad (\varepsilon\!\to\!0).
\]

\paragraph{Moving control volumes.}
If $\{\Omega_s\}$ is generated by a vector field $V$,
\[
\frac{\mathrm d}{\mathrm ds}\Big|_{s=0}\int_{\partial\Omega_s}\!\iota^{*}J
= \int_{\partial\Omega}\!\iota^{*}(\mathcal L_V J)
= \int_{\Omega}\!\mathcal L_V(\mathrm dJ) ,
\]
by Cartan’s formula (Reynolds transport, invariant form).
