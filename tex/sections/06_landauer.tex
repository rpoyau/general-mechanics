\section{Anchor Corollary I: Landauer Cost}
\label{sec:landauer}

\paragraph{Statement.}
In Kick units (Kk) with entropy in nats, erasing $\Delta S$ nats of Shannon information
in a reservoir at temperature $T$ requires the reversible work
\begin{equation}
\boxed{\, W_{\min} \;=\; T\,\Delta S \, }.
\end{equation}
If information is counted in bits ($\Delta H$), this reads $W_{\min}=T\,(\ln 2)\,\Delta H$.
In SI units: $W_{\min} = k_B\, T_{\mathrm K}\, \ln 2 \,\Delta H$ (see App.~\ref{sec:kick}).

\paragraph{Derivation from the resource–space identity.}
\begin{enumerate}
\item \textbf{Representation.} Choose the thermodynamic representation with
      $\rho=S$ (nats) and resources $R=(E,V,N_i,\dots)$.
      From the resource–space differential \eqref{eq:reos},
      \[
        d\rho + \Lambda_E\,dE + \Lambda_V\,dV - \sum_i \Lambda_{N_i}\,dN_i = 0,
      \]
      with dictionary $\Lambda_E=-\tfrac{1}{T}$, $\Lambda_V=-\tfrac{p}{T}$,
      $\Lambda_{N_i}=+\tfrac{\mu_i}{T}$.

\item \textbf{Isothermal reversible leg.} At fixed $(V,N_i)$ the Gibbs identity gives
      $dS=\tfrac{1}{T}\,dE$, hence $dE=T\,dS$.

\item \textbf{Integrate.} Reversible erasure from $S_1$ to $S_2$ yields
      $\Delta E_{\min}=\int T\,dS = T\,\Delta S$.
\end{enumerate}

\paragraph{Interpretation.}
This is the \emph{reversible} (minimum) cost implied directly by \eqref{eq:reos} and the
isothermal Gibbs relation. Any inequality $W\ge W_{\min}$ requires an irreversibility
assumption (Tier~1) and is not used at the axiomatic level.

Landauer \cite{Landauer1961}; see also algebraic proofs such as \cite{ReebWolf2014}.
