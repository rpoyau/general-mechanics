\section{Motivation}\label{sec:motivation}


\paragraph{Aim.}
We seek a compact, model-first mechanics in which \emph{relations} are primary and all observable ``forces'' are read as \emph{informational gradients across boundaries}.
The framework works in $n$ spatial dimensions (static Gauss surfaces) and in $(n{+}1)$-dimensional spacetime (fixed worldtubes), with dynamics derived from counting rather than postulated energies.

\paragraph{Three pillars.}
\begin{enumerate}[label=(P\arabic*),itemsep=0.25em]
  \item \textbf{Relationism.} Model content is constituted by relations $R$ (geometry, connectivity, boundary conditions). Microstates are those consistent with $R$; their count $W(R)$ defines $S[R]=\ln W(R)$.
  \item \textbf{Information.} State is a belief over micro-moves; adequacy is compression under constraints (Shannon/MDL). Distinctions define density $i$ and flux $\bm j$; production $\sigma\ge0$ records creation/erasure of distinctions.
  \item \textbf{Geometric calculus.} All balances are boundary identities (generalized Stokes); dynamics use differential forms, keeping the main thread metric-free until a constitutive map is chosen.
\end{enumerate}

\paragraph{Master balance.}
With information $4$-current $\JJ=(i,\bm j)$ and production $\sigma$,
\begin{equation}\label{eq:worldtube}
\nabla_\mu J_I^\mu=\sigma
\quad\Longleftrightarrow\quad
\int_{\partial\Omega}\!*j=\int_{\Omega}\sigma\,\mathrm d^{\,n+1}x
\end{equation}
for any fixed worldtube $\Omega=V\times[t_0,t_1]$.
A static snapshot ($\partial_t i=0$) reduces to Gauss on $S_{n-1}$ and yields the universal inverse-power law $|\nabla\Phi_I|\propto r^{1-n}$ for any channel with $\bm j=\kappa\nabla\Phi_I$.

\paragraph{REOS: dynamics as prices of relations.}
Relational changes $R^a\mapsto R^a+\mathrm dR^a$ induce
\begin{equation}
\mathrm d\rho \;=\; -\,\Lambda_a\,\mathrm dR^a,\qquad
\Lambda_a(x):=\frac{\delta S}{\delta R^a(x)}\ \ (\text{intensities from counting}).
\end{equation}
This \emph{Relational Equation of State} organizes dynamics as informational prices paid for changing relations, without invoking energies.

\paragraph{Path-rate mechanics.}
Counting micro-moves gives an information rate $\mathcal R(q,\dot q,t)$.
Define $P=\partial_{\dot q}\mathcal R$ and $\Phi=P\!\cdot\!\dot q-\mathcal R$; the information Poincar\'e--Cartan form
\begin{equation}
\Theta_{\mathcal R}=P\!\cdot\mathrm dq-\Phi\,\mathrm dt
\end{equation}
generates the canonical strip equations by Stokes: $\dot q=\partial_P\Phi$, $\dot P=-\partial_q\Phi$.
Lagrangian/Hamiltonian structures are merely names for the convex pair $(\mathcal R,\Phi)$.

\paragraph{Calibrations and units.}
Measured fields are linear calibrations of the informational field $\bm F_I=-\nabla\Phi_I$:
$\bm E=c_Q\,\bm F_I$, $\bm g=c_M\,\bm F_I$, and $\bm j_{\rm th}=\kappa_T\,\beta^{-1}\bm F_I$.
In \emph{Kick} units (Kk\,=\,Hz), $\Mkk=mc^2/h$ and $\Tkk=k_BT/h$, so $k_B$ and $G$ leave the algebra once $c_Q,c_M,\kappa_T$ are fixed by one datum each.

\paragraph{Corollaries.}
Maxwell-type identities ($d\mathcal F=0$, $d\mathcal H=j_3$) and the metric response
$G_{\mu\nu}=\chi_M T_{\mu\nu}^{(\mathrm{Kk})}$ are chosen for covariance and locality;
the Newton/Coulomb inverse-square laws are the \emph{static} shadows of \cref{eq:worldtube}.
Finite counting predicts Poisson-limited field estimates and a tiny Unruh-linked stochastic acceleration, setting a principled decoherence floor.

(ii) a bit-gas where pressure is momentum-bit flux; (iii) an entropic-gravity screen with inverse-square mean and $1/\sqrt{N}$ noise; and (iv) a covariant worldtube with information electrodynamics.
