\subsection{RIM--R1 (Integrability)}
Let $U\subset\mathbb{R}^{m}$ be a resource patch with coordinates $R^{(a)}$.
Define the $1$-form $\alpha(R):=\Lambda_a(R)\,dR^{(a)}$.
Axiom: $d\alpha=0$ on $U$.

\subsection{REOS (identity)}
If $U$ is simply connected, there exists a scalar potential $\rho(R)$, unique up to an additive constant, such that
\begin{equation}
  d\rho \;+\; \Lambda_a(R)\,dR^{(a)} \;=\; 0 .
  \label{eq:reos}
\end{equation}

\subsection{Maxwell symmetry}
For all indices $a,b$,
\[
  \partial_{R^{(b)}}\Lambda_a(R) \;=\; \partial_{R^{(a)}}\Lambda_b(R) .
\]
