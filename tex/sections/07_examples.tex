\section{Examples (Charts and Discrete)}
\label{sec:examples}

\subsection{1+1 chart}
Let $(\tau,x)$ be a local chart with volume form $d\tau\wedge dx$.
Write
\[
  J \;=\; j^{\tau}\, dx \;-\; j^{x}\, d\tau,
  \qquad
  \Sigma \;=\; \sigma\, d\tau\wedge dx .
\]
Then \eqref{eq:stokes-balance} reads
\begin{equation}
  \partial_{\tau} j^{\tau} + \partial_{x} j^{x} \;=\; \sigma .
\end{equation}
For a rectangle $V=[\tau_1,\tau_2]\times[x_1,x_2]$,
\[
  \int_{x_1}^{x_2}\! j^{\tau}(\tau_2,x)\,dx
  - \int_{x_1}^{x_2}\! j^{\tau}(\tau_1,x)\,dx
  + \int_{\tau_1}^{\tau_2}\! j^{x}(\tau,x_2)\,d\tau
  - \int_{\tau_1}^{\tau_2}\! j^{x}(\tau,x_1)\,d\tau
  \;=\; \iint_{V}\sigma\, d\tau\,dx .
\]

\subsection{2+1 chart}
On $(\tau,x,y)$ with volume form $d\tau\wedge dx\wedge dy$, write
\[
  J \;=\; j^{\tau}\, dx\wedge dy \;-\; j^{x}\, d\tau\wedge dy \;+\; j^{y}\, d\tau\wedge dx,
  \qquad
  \Sigma \;=\; \sigma\, d\tau\wedge dx\wedge dy .
\]
Then
\begin{equation}
  \partial_{\tau} j^{\tau} + \partial_{x} j^{x} + \partial_{y} j^{y} \;=\; \sigma .
\end{equation}

\subsection{3+1 chart}
On $(\tau,x,y,z)$ with volume form $d\tau\wedge dx\wedge dy\wedge dz$, write
\[
  J \;=\; j^{\tau}\, dx\wedge dy\wedge dz
  - j^{x}\, d\tau\wedge dy\wedge dz
  + j^{y}\, d\tau\wedge dx\wedge dz
  - j^{z}\, d\tau\wedge dx\wedge dy,
\]
\[
  \Sigma \;=\; \sigma\, d\tau\wedge dx\wedge dy\wedge dz ,
\]
giving
\begin{equation}
  \partial_{\tau} j^{\tau} + \partial_{x} j^{x} + \partial_{y} j^{y} + \partial_{z} j^{z} \;=\; \sigma .
\end{equation}

\subsection{Discrete analogue (finite chain complex)}
Let $C_k$ be oriented $k$-cells with incidence matrix $B:C_n\to C_{n-1}$.
Assign a current $J_e$ to each $(n\!-\!1)$-cell $e$ and a source $\sigma_c$ to each $n$-cell $c$.
The balance on $n$-cells is
\begin{equation}
  B^{\mathsf T} J \;=\; \sigma .
\end{equation}
For a union $V$ of $n$-cells, internal faces cancel and
\[
  \sum_{e\subset \partial V} \!\mathrm{sign}(e,V)\, J_e
  \;=\;
  \sum_{c\subset V} \sigma_c .
\]
