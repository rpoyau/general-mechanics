\section{Tier‑1: Jacobson‑Type Relation}
\label{sec:tier1-jacobson}

\subsection{T1J1 (Local horizon setup)}
Let $g$ be a Lorentzian metric on a neighbourhood of $P\in\mathcal M$.
Let $\mathcal H$ be a patch of local causal horizon generated by a future‑directed
null congruence with tangent $k^{a}$ and affine parameter $\lambda$
($k^{a}\nabla_{a}\lambda=1$).
Let $\chi^{a}$ be the approximate boost Killing field vanishing at $P$ with surface gravity $\kappa_{\mathcal H}$.

\subsection{T1J2 (Temperature and entropy)}
Define the Unruh temperature $T_{\mathcal H}:=\kappa_{\mathcal H}/(2\pi)$.
Let $s_{A}>0$ be an area–entropy density and set $dS = s_{A}\, dA$ on $\mathcal H$.

\subsection{T1J3 (Clausius identity)}
With energy–momentum tensor $T_{ab}$,
\begin{equation}
  \delta Q \;=\; \int_{\mathcal H} T_{ab}\,\chi^{a}\, d\Sigma^{b},
  \qquad
  d\Sigma^{b} := k^{b}\, d\lambda\, dA ,
\end{equation}
impose $\delta Q = T_{\mathcal H}\, dS$ for all local horizon patches.

\subsection{T1J4 (Field equation)}
Using the Raychaudhuri equation for $k^{a}$ and equilibrium initial data
$\theta|_{P}=0$, $\sigma_{ab}|_{P}=0$,
\begin{equation}
  R_{ab} - \tfrac12 R\, g_{ab} + \Lambda\, g_{ab}
  \;=\; \kappa_{J}\, T_{ab},
  \label{eq:jacobson-einstein}
\end{equation}
with constants $\kappa_{J}>0$ and $\Lambda$ determined by $s_{A}$ and units.

\subsection{T1J5 (Thermodynamic identification)}
Along horizon flux with energy exchange only,
\begin{equation}
  dS \;=\; \frac{1}{T_{\mathcal H}}\, dE ,
  \label{eq:jacobson-gibbs}
\end{equation}
matching the resource–space identity with $\Lambda_E=-1/T_{\mathcal H}$.
