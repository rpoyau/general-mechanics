\section{Differential forms \& generalized Stokes}
\label{app:forms-stokes}

\subsection{Manifolds, orientation, and volume forms}
\label{app:forms-stokes:manifolds}
Let \(M\) be an \((n{+}1)\)-dimensional oriented manifold with local coordinates \((x^1,\dots,x^n,t)\).
Fix a region \(U\subset M\) with smooth spatial \emph{frames} \(V_t:=U\cap\{t=\mathrm{const}\}\) whose boundary is \(\partial V_t\).
The product region \(U_{[t_0,t_1]}:=U\cap\{t\in[t_0,t_1]\}\) has oriented boundary
\[
\partial U_{[t_0,t_1]}
=\underbrace{V_{t_1}}_{\text{top cap}}
\;\cup\;
\underbrace{(-V_{t_0})}_{\text{bottom cap}}
\;\cup\;
\underbrace{\big(\partial V\times[t_0,t_1]\big)}_{\text{side}}.
\]
Write \(\mathrm{vol}_{n}\) for the \(n\)-volume form on \(V_t\) and \(\mathrm{vol}_{n+1}\) for the \((n{+}1)\)-volume form on \(U\).

\subsection{Current 1‑form, Hodge dual, and production}
\label{app:forms-stokes:current}
Let \(j:=J^\mu dx_\mu\) denote a current \(1\)-form on \(U\) with components \(J^\mu=(i,\mathbf j)\) in adapted
coordinates. Its Hodge dual \(j^{(n)}:=\!*j\) is an \(n\)-form that integrates over spatial caps and side surfaces.
Production is the \((n{+}1)\)-form \(\Sigma:=\sigma\,\mathrm{vol}_{n+1}\) with density \(\sigma\).

\subsection{Global identity (generalized Stokes)}
\label{app:forms-stokes:global}
The Stokes identity reads
\begin{equation}
d\,j^{(n)}=\Sigma
\quad\Longleftrightarrow\quad
\partial_\mu J^\mu=\sigma .
\label{app:forms-stokes:eq}
\end{equation}
Equivalently, for any product region \(U_{[t_0,t_1]}\),
\begin{equation}
\int_{\partial U_{[t_0,t_1]}} j^{(n)} \;=\; \int_{U_{[t_0,t_1]}} \Sigma .
\label{app:forms-stokes:eq:int}
\end{equation}

\subsection{Cap/side decomposition and the cut identity}
\label{app:forms-stokes:cap-side}
Using the decomposition of \(\partial U_{[t_0,t_1]}\),
\begin{align}
\int_{\partial U_{[t_0,t_1]}}\!\! j^{(n)}
&=
\underbrace{\int_{V_{t_1}}\! j^{(n)}}_{\text{top cap}}
-
\underbrace{\int_{V_{t_0}}\! j^{(n)}}_{\text{bottom cap}}
+
\underbrace{\int_{\partial V\times[t_0,t_1]}\!\! j^{(n)}}_{\text{side}} .
\label{app:forms-stokes:eq:cap-side}
\end{align}
Define
\[
I_{\mathrm{enc}}(t):=\int_{V_t} i\,\mathrm{vol}_{n},\qquad
\Phi_{\mathrm{in/out}}(t):=\int_{\partial V_t}\!\big(\mathbf j\!\cdot\!\mathbf n\big)_{\mp}\,\mathrm dA,
\qquad
\Pi(t):=\int_{V_t}\sigma\,\mathrm d^{n}x .
\]
Differentiating \eqref{app:forms-stokes:eq:int} in \(t\) yields the cut identity
\begin{equation}
\dot I_{\mathrm{enc}}(t)\;=\;\Phi_{\mathrm{in}}(t)-\Phi_{\mathrm{out}}(t)\;+\;\Pi(t).
\label{app:forms-stokes:eq:cut}
\end{equation}

\paragraph{Worldtube (moving boundaries).}
If the boundary moves with normal speed \(v_n\), the side integrand is \(\mathbf j\!\cdot\!\mathbf n - i\,v_n\);
caps are evaluated on the initial and final frames. This is the form used when \(A_{n}(r,t)\) varies in time.

\subsection{Local form (continuity)}
\label{app:forms-stokes:local}
In adapted coordinates \((x,t)\) the differential statement \eqref{app:forms-stokes:eq} yields
\begin{equation}
\partial_t i(x,t)\;+\;\nabla\!\cdot \mathbf j(x,t)\;=\;\sigma(x,t),
\label{app:forms-stokes:eq:local}
\end{equation}
the usual \((n{+}1)\)-dimensional continuity law.

\subsection{Frame and tube/boundary invariants}
\label{app:forms-stokes:frame}
On a \frame, on an open set \(W\subset U\) with \(\sigma|_W\equiv 0\) (source‑free),
\(d j^{(n)}=0\) on \(W\) implies that the flux through any two homologous boundaries in \(W\) is equal.
If \(\mathcal T\) is a narrow tube whose side surface is tangent to \(\mathbf j\), then the directed tube flux
\(\Phi=\int_{\Sigma}\mathbf j\!\cdot\!\mathbf n\,\mathrm dA\) is constant along the tube and cross–sectional averages take the form
\[
\langle j_n\rangle=\frac{\Phi}{A}.
\]

\subsection{Conservation (window statement)}
\label{app:forms-stokes:poynting}
On a finite time window, one may write a local balance in the neutral form
\begin{equation}
\partial_t u\;+\;\nabla\!\cdot\mathbf S\;=\;-\,\Pi,\qquad \Pi\ge 0,
\label{app:forms-stokes:eq:conservation-local}
\end{equation}
where \(u\) is an energy‑like density, \(\mathbf S\) a spatial flux, and \(\Pi\) a production rate.
Integrating over a frame \(V_t\) yields
\begin{equation}
\frac{\mathrm d}{\mathrm dt}\int_{V_t}\!u\,\mathrm d^n x
\;+\;
\int_{\partial V_t}\!\mathbf S\!\cdot\!\mathbf n\,\mathrm dA
\;=\;
-\int_{V_t}\!\Pi\,\mathrm d^n x .
\label{app:forms-stokes:eq:conservation}
\end{equation}
In a lossless/source‑free subregion (\(\Pi=0\)), the exported quantity \(\int_{\partial V_t}\mathbf S\!\cdot\!\mathbf n\,\mathrm dA\) is equal across homologous boundaries.
\emph{Frame statements are the \(\Delta t\!\to\!0\) limit of \eqref{app:forms-stokes:eq:conservation}.}
Closure‑specific realizations appear in \S\ref{waves}.

\subsection{Sign conventions and directional split}
\label{app:forms-stokes:signs}
Reversing the orientation of \(V_t\) or the time direction flips signs consistently in
\eqref{app:forms-stokes:eq:cap-side}–\eqref{app:forms-stokes:eq:local}. The inward/outward split is defined by
\((a)_\pm:=\tfrac12\,(a\pm|a|)\) and is independent of those choices.

\subsection{Summary}
\label{app:forms-stokes:summary}
With \(j\) a \(1\)-form and \(j^{(n)}=\!*j\), the single identity \(d j^{(n)}=\sigma\,\mathrm{vol}_{n+1}\) yields:
(i) the local continuity equation \eqref{app:forms-stokes:eq:local}, and (ii) the cut identity
\eqref{app:forms-stokes:eq:cut} from the cap/side decomposition of \(\partial U_{[t_0,t_1]}\).
Exported‑power statements are read only after a closure is specified elsewhere; see \S\ref{waves}.
