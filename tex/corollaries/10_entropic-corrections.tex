\subsection{Entropic–gravity corrections (attenuation, production, capacity)}
\label{corollary:entropic-corrections}

This corollary derives \emph{frame} corrections to the inverse–area law when the region is not perfectly
lossless (attenuation/production present) and/or when boundary counting saturates near the source.
All statements follow from the Stokes identity \eqref{axioms:stokes:identity:eq} and the \emph{frame}
linear closure \eqref{axioms:stokes:frame:eq:closure}; no new constants are introduced.

\paragraph{Flux from Stokes on a spherical boundary (general \(n\)).}
Let \(A_n(r)=S_{n-1}\,r^{\,n-1}\) be the area of a radius–\(r\) boundary in \(n\) spatial dimensions,
\(j_r(r)\) the radial \emph{flux density}, and \(\Phi(r):=A_n(r)\,j_r(r)\) the directed flux through the boundary.
On a frame, the continuity law gives
\begin{equation}
\frac{d}{dr}\,\Phi(r)\;=\;A_n(r)\,\sigma(r).
\label{corollary:entropic-corrections:eq:dphi}
\end{equation}
With the frame closure \(j_r=-\kappa\,E(r)\), where \(E(r):=|\mathbf E_I(r)|\),
\begin{equation}
\frac{d}{dr}\Bigl(A_n(r)\,E(r)\Bigr)\;=\;-\frac{1}{\kappa}\,A_n(r)\,\sigma(r).
\label{corollary:entropic-corrections:eq:core}
\end{equation}

\paragraph{Attenuation model (collisional sink).}
Suppose interactions remove directed flux at a local rate \(\mu(r)\ge 0\) per unit length.
The sink is \(\sigma_{\rm sink}(r)=-\mu(r)\,j_r(r)\), hence \(A_n\sigma_{\rm sink}=-\mu\,\Phi\).
Using \(j_r=-\kappa E\) in \eqref{corollary:entropic-corrections:eq:core} yields the linear ODE
\begin{equation}
\boxed{\quad \frac{d}{dr}\Bigl(A_n E\Bigr) + \mu(r)\,A_n\,E\;=\;0 \quad}
\label{corollary:entropic-corrections:eq:ode-atten}
\end{equation}
with solution
\begin{equation}
E(r)\;=\;\frac{C}{A_n(r)}\,
\exp\!\Bigl(\!-\,\int_{r_0}^{r}\!\mu(u)\,du\Bigr),
\qquad C=\frac{\Phi(r_0)}{\kappa}.
\label{corollary:entropic-corrections:eq:sol-atten}
\end{equation}
Equivalently, along a tube coordinate \(s\) one has the survival factor \(S=\exp\!\bigl[-\int\mu(s)\,ds\bigr]\);
for radial frames \(ds=dr\).

\paragraph{Examples (3D).}
For \(n=3\), \(A_3(r)=4\pi r^2\):
\begin{equation}
E(r)=\frac{C}{4\pi r^2}\,\exp\!\Bigl(-\!\int_{r_0}^{r}\mu(u)\,du\Bigr).
\label{corollary:entropic-corrections:eq:3d-general}
\end{equation}
Special cases:
\[
\mu(r)=0 \Rightarrow E=\frac{C}{4\pi r^2};\qquad
\mu(r)=\mu_0 \Rightarrow E=\frac{C}{4\pi r^2}\,e^{-\mu_0(r-r_0)};
\qquad
\mu(r)=\frac{a}{r} \Rightarrow E=\frac{C}{4\pi}\,\frac{r_0^{a}}{r^{2+a}}.
\]

\paragraph{Volumetric source/sink (general forcing).}
If, in addition, there is a bounded source density \(\sigma_{\rm src}(r)\), then
\(\sigma=\sigma_{\rm src}-\mu\,j_r\) and \eqref{corollary:entropic-corrections:eq:core} becomes
\begin{equation}
\frac{d}{dr}\bigl(A_n E\bigr)+\mu(r)\,A_n\,E
\;=\;-\frac{1}{\kappa}\,A_n(r)\,\sigma_{\rm src}(r).
\label{corollary:entropic-corrections:eq:ode-forced}
\end{equation}
The solution is the integrating–factor form
\begin{equation}
E(r)=\frac{1}{A_n(r)}\,e^{-\!\int_{r_0}^{r}\mu}\!
\left[
C - \frac{1}{\kappa}\!
\int_{r_0}^{r}\! A_n(\rho)\,\sigma_{\rm src}(\rho)\,
e^{\int_{r_0}^{\rho}\mu}\, d\rho
\right].
\label{corollary:entropic-corrections:eq:sol-forced}
\end{equation}

\paragraph{Capacity/\emph{trit}–density ceiling (inner flattening).}
Let the boundary at radius \(r\) admit \(N(r)\) addressable \emph{trits} and a per–trit reliable rate \(f_{\max}\).
Then the inward counting rate satisfies \(J_A(r)\le f_{\max}N(r)\) so that
\(j_r(r)\le f_{\max}N(r)/A_n(r)\).
With a bookkeeping model \(N(r)=\alpha\,A_n(r)\) (trits), one obtains a constant ceiling
\begin{equation}
E(r)\ \le\ \frac{f_{\max}\alpha}{\kappa}\;=:\;E_{\max}.
\label{corollary:entropic-corrections:eq:emax}
\end{equation}
In \(n=3\) the unattenuated form \(E=C/(4\pi r^2)\) intersects the ceiling at
\begin{equation}
r_{\rm flat}\;=\;\sqrt{\frac{C}{4\pi\,E_{\max}}}\,,
\qquad
E(r)\;=\;\min\!\left\{\ \frac{C}{4\pi r^2}\,e^{-\!\int_{r_0}^{r}\mu},\ \ E_{\max}\ \right\}.
\label{corollary:entropic-corrections:eq:rflat}
\end{equation}
Inside \(r_{\rm flat}\) the profile flattens; outside, it follows the (possibly attenuated) inverse–square law.

\paragraph{Piecewise media and composition.}
For piecewise–constant segments \(\{(\mu_k,\sigma_{{\rm src},k},r_k\!\to\!r_{k+1})\}\), multiply survival
factors \(e^{-\mu_k \Delta r_k}\) and add forced contributions segment by segment using
\eqref{corollary:entropic-corrections:eq:sol-forced}. The ceiling \eqref{corollary:entropic-corrections:eq:emax}
may be imposed per segment if \(N(r)\) or \(f_{\max}\) changes.

\paragraph{Potential drop.}
Since \(\mathbf E_I=-\nabla\Phi_I\), the potential difference between radii \(r_1<r_2\) is
\begin{equation}
\Phi_I(r_2)-\Phi_I(r_1) \;=\; -\int_{r_1}^{r_2}\! E(\rho)\, d\rho .
\label{corollary:entropic-corrections:eq:potential}
\end{equation}
With \eqref{corollary:entropic-corrections:eq:3d-general} this integrates to elementary functions for constant
or power–law \(\mu(r)\).

\paragraph{Remarks and limits.}
\begin{itemize}\itemsep2pt
\item \emph{Boundary sector/tube form.} For a boundary sector of solid angle \(\Delta\Omega\) replace \(A_n(r)\)
by \(A_\Delta(r)=\Delta\Omega\,r^{\,n-1}\) in
\eqref{corollary:entropic-corrections:eq:ode-atten}–\eqref{corollary:entropic-corrections:eq:sol-forced}.
\item \emph{Confinement.} In effective 1D (guides) or 2D (sheets) geometries, replace \(A_n(r)\) by the
constant cross–section or circumference; the same ODE applies and removes geometric spreading
(cf.\ \S\ref{corollary:dimensional}).
\item \emph{Operational link to time–information.} When durable records are formed and later erased, the available
rate is also limited by \(P/T\); the ceiling \eqref{corollary:entropic-corrections:eq:emax} and the
time–information bounds in \S\ref{corollary:time-info} are complementary constraints.
\end{itemize}

\medskip
\noindent\emph{Summary.}
\emph{Frame} fields on boundaries satisfy a first–order linear ODE for \(A_n(r)E(r)\).
Attenuation multiplies the inverse–power law by a survival factor \(e^{-\!\int\mu}\); volumetric sources
superpose through \eqref{corollary:entropic-corrections:eq:sol-forced}; and finite counting capacity (in trits)
imposes an inner ceiling \(E_{\max}\), producing a flattening radius \(r_{\rm flat}\). These results are constant–free.
