\subsection{Continuity \& Clausius}
\label{corollary:continuity}

\paragraph{Setup.}
Let \(V\subset U\) be a fixed region with outward unit normal \(\mathbf n\).
On a boundary patch \(A\subset\partial V\) over a \emph{window} (\S\ref{axioms:window}),
denote the boundary entropy flux density by \(\mathbf j_S\).

\paragraph{Boundary heat.}
On \([t_0,t_1]\),
\begin{equation}
Q_A\!\left([t_0,t_1]\right)
\;=\;
\int_{t_0}^{t_1}\!\!\int_A T\,\mathbf j_S\!\cdot\!\mathbf n\,\mathrm dA\,\mathrm dt,
\label{corollary:continuity:eq:total-heat}
\end{equation}
and the frame heat‑flux density is
\begin{equation}
\mathbf q \;=\; T\,\mathbf j_S .
\label{corollary:continuity:eq:heat-flux}
\end{equation}

\paragraph{Clausius form (local).}
With \(s=i\) and \(\mathbf j_S=\mathbf q/T\) (from \S\ref{axioms:boundary-heat}), the continuity law
\eqref{axioms:stokes:identity:eq:local-continuity} gives
\begin{equation}
\partial_t s \;+\; \nabla\!\cdot\!\left(\frac{\mathbf q}{T}\right) \;=\; \sigma \ \ge\ 0 .
\label{corollary:continuity:eq:clausius-local}
\end{equation}

\paragraph{Clausius form (cut).}
From \eqref{axioms:stokes:identity:eq:cut}, using the directional split \S\ref{axioms:directional-split},
\begin{equation}
\dot S_{\mathrm{enc}}
\;=\;
\int_{\partial V}\frac{\big(\mathbf q\!\cdot\!\mathbf n\big)_-}{T}\,\mathrm dA
\;-\;
\int_{\partial V}\frac{\big(\mathbf q\!\cdot\!\mathbf n\big)_+}{T}\,\mathrm dA
\;+\;
\Pi_{\mathrm th}
\ \ \ge\ 0 ,
\label{corollary:continuity:eq:clausius-cut}
\end{equation}
with \(\Pi_{\mathrm th}:=\int_V \sigma\,\mathrm d^n x\).

\paragraph{Calibration to the informational field.}
With the Linear Channel Closure \eqref{axioms:linear-closure:eq} and the one–datum sector calibration,
\begin{equation}
\mathbf j \;=\; -\,\kappa\,\mathbf E_I,
\qquad
\mathbf q \;=\; -\,T\,\kappa\,\mathbf E_I .
\label{corollary:continuity:eq:calibrated}
\end{equation}
