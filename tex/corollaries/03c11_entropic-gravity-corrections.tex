\subsection{C11: Entropic–gravity corrections (attenuation, production, capacity)}
\label{corollary:entropic-corrections}
This corollary derive static corrections to the inverse–area law when the channel is not perfectly lossless
and/or when counting saturates near the source. All statements follow from the (generalized) Stokes
identity \eqref{eq:stokes-identity} and the static closure \eqref{eq:static-closure}; no new constants
are introduced.

\paragraph{Flux from Stokes on a spherical screen (general $n$).}
Let $A_n(r)=S_{n-1}\,r^{\,n-1}$ be the area of a radius–$r$ screen in $n$ spatial dimensions,
$j_r(r)$ the radial current, and $\Phi(r):=A_n(r)\,j_r(r)$ the directed flux through the screen.
For static fields, the continuity law gives
\begin{equation}
\frac{d}{dr}\,\Phi(r)\;=\;A_n(r)\,\sigma(r).
\label{eq:c11-dphi}
\end{equation}
With the static closure $j_r=-\kappa\,E(r)$, where $E(r):=|\mathbf E_I(r)|$,
\begin{equation}
\frac{d}{dr}\bigl(A_n(r)\,E(r)\bigr)\;=\;-\frac{1}{\kappa}\,A_n(r)\,\sigma(r).
\label{eq:c11-core}
\end{equation}

\paragraph{Attenuation model (collisional sink).}
Suppose interactions remove directed flux at a local rate $\lambda(r)\ge 0$ per unit length.
The sink is $\sigma_{\rm sink}(r)=-\lambda(r)\,j_r(r)$, hence $A_n\sigma_{\rm sink}=-\lambda\,\Phi$.
Using $j_r=-\kappa E$ in \eqref{eq:c11-core} yields the linear ODE
\begin{equation}
\boxed{\quad \frac{d}{dr}\bigl(A_n E\bigr) + \lambda(r)\,A_n\,E\;=\;0 \quad}
\label{eq:c11-ode-atten}
\end{equation}
with solution
\begin{equation}
E(r)\;=\;\frac{C}{A_n(r)}\,
\exp\!\Bigl(\!-\,\int_{r_0}^{r}\!\lambda(u)\,du\Bigr),
\qquad C=\frac{\Phi(r_0)}{\kappa}.
\label{eq:c11-sol-atten}
\end{equation}

\paragraph{Examples (3D).}
For $n=3$, $A_3(r)=4\pi r^2$:
\begin{equation}
E(r)=\frac{C}{4\pi r^2}\,e^{-\int_{r_0}^{r}\lambda(u)\,du}.
\label{eq:c11-3d-general}
\end{equation}
Special cases:
\[
\lambda(r)=0 \Rightarrow E=\frac{C}{4\pi r^2};\qquad
\lambda(r)=\lambda_0 \Rightarrow E=\frac{C}{4\pi r^2}\,e^{-\lambda_0(r-r_0)};
\qquad
\lambda(r)=\frac{a}{r} \Rightarrow E=\frac{C}{4\pi}\,\frac{r_0^{a}}{r^{2+a}}.
\]

\paragraph{Volumetric source/sink (general forcing).}
If, in addition, there is a bounded source density $\sigma_{\rm src}(r)$, then
$\sigma=\sigma_{\rm src}-\lambda\,j_r$ and \eqref{eq:c11-core} becomes
\begin{equation}
\frac{d}{dr}\bigl(A_n E\bigr)+\lambda(r)\,A_n\,E
\;=\;-\frac{1}{\kappa}\,A_n(r)\,\sigma_{\rm src}(r).
\label{eq:c11-ode-forced}
\end{equation}
The solution is the integrating–factor form
\begin{equation}
E(r)=\frac{1}{A_n(r)}\,e^{-\!\int_{r_0}^{r}\lambda}\!
\left[
C - \frac{1}{\kappa}\!
\int_{r_0}^{r}\! A_n(\rho)\,\sigma_{\rm src}(\rho)\,
e^{\int_{r_0}^{\rho}\lambda}\, d\rho
\right].
\label{eq:c11-sol-forced}
\end{equation}

\paragraph{Capacity/\emph{trit}–density ceiling (inner flattening).}
Let the screen at radius $r$ admit $N(r)$ addressable \emph{trits} and a per–trit reliable rate $f_{\max}$.
Then the inward counting rate satisfies $J_A(r)\le f_{\max}N(r)$ so that
$j_r(r)\le f_{\max}N(r)/A_n(r)$. With a bookkeeping model $N(r)=\alpha\,A_n(r)$ (trits), one obtains a constant ceiling
\begin{equation}
E(r)\ \le\ \frac{f_{\max}\alpha}{\kappa}\;=:\;E_{\max}.
\label{eq:c11-emax}
\end{equation}
In $n=3$ the unattenuated form $E=C/(4\pi r^2)$ intersects the ceiling at
\begin{equation}
r_{\rm flat}\;=\;\sqrt{\frac{C}{4\pi\,E_{\max}}}\,,
\qquad
E(r)\;=\;\min\!\left\{\ \frac{C}{4\pi r^2}\,e^{-\!\int_{r_0}^{r}\lambda},\ \ E_{\max}\ \right\}.
\label{eq:c11-rflat}
\end{equation}
Inside $r_{\rm flat}$ the profile flattens; outside, it follows the (possibly attenuated) inverse–square law.

\paragraph{Piecewise media and composition.}
For piecewise–constant segments $\{(\lambda_k,\sigma_{{\rm src},k},r_k\!\to\!r_{k+1})\}$, multiply survival
factors $e^{-\lambda_k \Delta r_k}$ and add forced contributions segment by segment using
\eqref{eq:c11-sol-forced}. The ceiling \eqref{eq:c11-emax} may be imposed per segment if $N(r)$ or $f_{\max}$ changes.

\paragraph{Potential drop.}
Since $\mathbf E_I=-\nabla\Phi_I$, the potential difference between radii $r_1<r_2$ is
\begin{equation}
\Phi_I(r_2)-\Phi_I(r_1) \;=\; -\int_{r_1}^{r_2}\! E(\rho)\, d\rho .
\label{eq:c11-potential}
\end{equation}
With \eqref{eq:c11-3d-general} this integrates to elementary functions for constant or power–law $\lambda(r)$.

\paragraph{Remarks and limits.}
\begin{itemize}\itemsep2pt
\item \emph{Sector/tube form.} For an aperture of solid angle $\Delta\Omega$ replace $A_n(r)$ by $A_\Delta(r)=\Delta\Omega\,r^{\,n-1}$ in
\eqref{eq:c11-ode-atten}–\eqref{eq:c11-sol-forced}.
\item \emph{Confinement.} In effective 1D (guides) or 2D (sheets) geometries, replace $A_n(r)$ by the
constant cross–section or circumference; the same ODE applies and removes geometric spreading
(cf.\ C8, \S\ref{corollary:dimensional}).
\item \emph{Operational link to C5.} When durable records are formed and later erased, the available rate is also limited
by $P/T$; the ceiling \eqref{eq:c11-emax} and the time–information bounds in C5 are complementary constraints
(\S\ref{corollary:time-info}).
\end{itemize}

\medskip
\noindent\emph{Summary.}
Static fields on screens satisfy a first–order linear ODE for $A_n(r)E(r)$. Attenuation multiplies the
inverse–power law by a survival factor $e^{-\!\int\lambda}$; volumetric sources superpose through
\eqref{eq:c11-sol-forced}; and finite counting capacity (in trits) imposes an inner ceiling $E_{\max}$,
producing a flattening radius $r_{\rm flat}$. These results are constant–free.
