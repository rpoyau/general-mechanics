\subsection{C12: Quantum path–integral template (from informational fluctuations)}
\label{corollary:qpi-template}

\paragraph{Scope and stance.}
This rung records a \emph{template} for wave–like evolution that can be overlaid on the framework
when one accepts a quadratic local model for small fluctuations of boundary counting.
It is \emph{not} an axiom. The canonical drift remains fixed by the rate 1‑form and its closure.

\paragraph{Canonical rates (recap).}
From Axiom~D4, the Poincar\'e--Cartan (rate) 1‑form and closure,
\begin{equation}
\Theta_{\mathcal R}(q,P,t)=P_i\,dq^i-H(q,P,t)\,dt,
\qquad
d\Theta_{\mathcal R}=0,
\label{eq:c12-pc-form}
\end{equation}
yield the canonical system
\begin{equation}
\dot q^i=\partial_{P_i}H,\qquad \dot P_i=-\,\partial_{q^i}H ,
\label{eq:c12-canonical}
\end{equation}
identical in content to \eqref{eq:pc-form}–\eqref{eq:canonical}.

\paragraph{Diffusive branch (MaxCal/Onsager--Machlup).}
On a window where local fluctuations around the canonical drift are modeled by a Gaussian kernel
with positive‑definite diffusivity matrix $D(q,t)$ and a scalar rate field $U(q,t)$,
maximizing path caliber with constraints on $\int |\dot q-v|_{D^{-1}}^2\,dt$ and $\int U\,dt$
yields the Onsager--Machlup rate
\begin{equation}
\mathcal R_{\rm diff}(q,\dot q,t)
=\frac{1}{2}\,(\dot q-v)^T D^{-1}(\dot q-v)+U(q,t),
\label{eq:c12-OM}
\end{equation}
and the path weight
\begin{equation}
\mathbb P[\gamma]\ \propto\ 
\exp\!\left(-\int_{t_0}^{t_1}\mathcal R_{\rm diff}\,dt\right).
\label{eq:c12-weight}
\end{equation}
The density $\rho(q,t)$ then takes a Fokker--Planck form
\begin{equation}
\partial_t \rho
=
-\nabla\!\cdot(v\,\rho)
+\tfrac12\,\nabla^T\!\bigl(D\,\nabla\rho\bigr)
-\bigl(\nabla U\bigr)\!\cdot\!\partial_{\dot q}\rho
\quad \text{(as modeled)},
\label{eq:c12-fp}
\end{equation}
with $D$ tied to the counting noise kernel of App.~\ref{app:noise-kernel} (see C6, \S\ref{corollary:noise}).

\paragraph{Oscillatory branch (wave template).}
Keep the same local fields but rotate the quadratic kernel to an \emph{oscillatory} weight:
\begin{equation}
\mathcal A[\gamma]\ \propto\
\exp\!\left(i\,2\pi\!\int_{t_0}^{t_1}\!\Big[P\!\cdot\!\dot q-H(q,P,t)\Big]\,dt\right).
\label{eq:c12-amp}
\end{equation}
Choose a quadratic generator with an inertial map $M(q,t)$:
\begin{equation}
H(q,P,t)=\tfrac12\,P^T M^{-1}(q,t)\,P+U(q,t),
\label{eq:c12-quadH}
\end{equation}
so that stationary phase reproduces the canonical drift \eqref{eq:c12-canonical} with
$\dot q=M^{-1}P$ and $\dot P=-\nabla U$ (cf.\ C9, \S\ref{corollary:thermo-newton} for the Newtonian specialization).

\paragraph{Short‑time kernel and a Schr\"odinger‑type evolution (template).}
Under the quadratic choice \eqref{eq:c12-quadH}, a Fourier representation of the short‑time kernel
for \eqref{eq:c12-amp} yields a linear evolution for a complex amplitude $\psi(q,t)$:
\begin{equation}
i\,\partial_t \psi
=
\left[
-\frac{\alpha^{2}}{2}\,\nabla^T\!\bigl(M^{-1}\nabla\bigr)
\ +\ U(q,t)
\right]\psi,
\qquad \alpha>0,
\label{eq:c12-schro}
\end{equation}
where $\alpha$ is a positive wave scale and $M$ an inertial map (both fixed once per sector when used).
No universal constant is introduced.

\paragraph{Minimal coupling (Info--EM specialization).}
If an informational 1‑form $A_I$ is present (C4, \S\ref{corollary:info-em}), minimal coupling enters by
\begin{equation}
P\ \mapsto\ P-A_I(q,t)
\quad\text{in}\quad
\eqref{eq:c12-amp}–\eqref{eq:c12-schro},
\label{eq:c12-min-coupling}
\end{equation}
preserving the Poincar\'e--Cartan structure and producing gauge‑covariant drift and waves.
(A sector charge matrix may be included as a modeling choice.)

\paragraph{Relations between the branches.}
Both closures reduce to the same canonical drift \eqref{eq:c12-canonical} under stationary phase.
Heuristically, larger counting noise (bigger $D$ in C6/App.~\ref{app:noise-kernel}) narrows resolvable interference,
effectively restricting admissible $\alpha$ in \eqref{eq:c12-schro}. The branch choice is empirical
and channel‑dependent.

\paragraph{Units.}
Rates follow the global convention in App.~\ref{app:notation-units} (Kz; see \S\ref{appH:kz}).
Any sector‑specific scales ($M$, $\alpha$) are fixed once when this template is used; no bare constants are carried by the algebra.

\medskip
\noindent\emph{Summary.}
The Stokes identity on $(q,t)$ fixes the canonical drift. Adding a quadratic local kernel yields two
consistent fluctuation closures: a diffusive (Fokker--Planck) branch and, by oscillatory continuation,
a Schr\"odinger‑type template. Both sit on the same manifold first principles and do not introduce
bare constants.

\medskip
\noindent\emph{Literature note.}
Diffusive (Onsager--Machlup) and Fokker--Planck treatments are standard \cite{OnsagerMachlup1953,Risken1989,Gardiner2004}.
The oscillatory path‑integral template follows the classic construction \cite{Feynman1948,FeynmanHibbs1965}.
The stochastic/noise‑kernel connection is summarized in App.~\ref{app:noise-kernel}; see also \cite{HuVerdaguer2008} for a broad review context.
