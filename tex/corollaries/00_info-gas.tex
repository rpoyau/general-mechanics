\subsection{Info–Gas \& Stokes}
\label{corollary:info-gas}

\paragraph{Setup.}
A narrow tube \(\mathcal T\) has side tangent to \(\mathbf j\) and cross‑sections \(\Sigma(r)\) orthogonal to its axis.
Work on a frame with \(\partial_t i=0\); the tube lies in an open \(W\subset U\) with \(\sigma|_W=0\).

\paragraph{Derivation.}
From the local continuity identity \(\eqref{axioms:stokes:identity:eq:local-continuity}\) on \(W\), the directed flux through \(\Sigma(r)\) is independent of \(r\):
\begin{equation}
\Phi(r):=\int_{\Sigma(r)} \mathbf j\!\cdot\!\mathbf n\,\mathrm dA \;=\; \Phi \;\;\text{(const.)}.
\label{corollary:info-gas:eq:flux-const}
\end{equation}
Hence the cross‑sectional mean of the normal component satisfies
\begin{equation}
\big\langle j_n\big\rangle(r)\;=\;\frac{\Phi}{A(r)},\qquad
A(r):=\int_{\Sigma(r)} \mathrm dA .
\label{corollary:info-gas:eq:primary}
\end{equation}

\paragraph{Specialization.}
For a conical sector of fixed solid angle \(\Delta\Omega\) in \(n\) spatial dimensions,
\begin{equation}
A_n(r)=\Delta\Omega\,r^{\,n-1}
\;\Rightarrow\;
\big\langle j_n\big\rangle(r)=\frac{\Phi}{\Delta\Omega}\,r^{-(n-1)} .
\label{corollary:info-gas:eq:special}
\end{equation}

\paragraph{Calibration.}
With the Linear Channel Closure \(\eqref{axioms:linear-closure:eq}\),
\begin{equation}
\big|\mathbf E_I(r)\big|=\frac{1}{\kappa}\,\big\langle j_n\big\rangle(r)
=\frac{1}{\kappa}\,\frac{\Phi}{A(r)} .
\label{corollary:info-gas:eq:calib}
\end{equation}

\paragraph{Falsification.}
On a source‑free tube, if \(\big\langle j_n\big\rangle(r)\,A(r)\) deviates from a constant by more than a stated tolerance over a declared \(r\)‑range, the premises of this corollary are contradicted.

\(\eqref{corollary:info-gas:eq:primary}\).
