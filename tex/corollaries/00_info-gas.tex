\subsection{Info–Gas \& Stokes}
\label{corollary:info-gas}

\paragraph{Setup.}
A narrow tube \(\mathcal T\) has side tangent to \(\mathbf j\) and cross‑sections \(\Sigma(r)\) orthogonal to its axis.
On a \frame (\S\ref{axioms:frame}) \(\partial_t i=0\) and on an open set \(W\subset U\) with \(\sigma|_W=0\), the tube lies in \(W\).

\paragraph{Tube invariance.}
For
\begin{equation}
\Phi(r):=\int_{\Sigma(r)} \mathbf j\!\cdot \mathbf n\,\mathrm dA ,
\label{corollary:info-gas:eq:flux}
\end{equation}
the local continuity law \eqref{axioms:stokes:identity:eq:local-continuity} on \(W\) implies
\begin{equation}
\Phi(r)=\mathrm{const},
\qquad
\big\langle j_n\big\rangle(r)=\frac{\Phi}{A(r)},
\label{corollary:info-gas:eq:tube}
\end{equation}
with \(A(r):=\int_{\Sigma(r)}\mathrm dA\) and \(\langle j_n\rangle\) the cross‑sectional mean of the normal component.

\paragraph{Cones \& boundary sectors.}
For a conical sector of fixed solid angle \(\Delta\Omega\) in \(n\) spatial dimensions,
\begin{equation}
A_n(r)=\Delta\Omega\,r^{\,n-1}
\;\Rightarrow\;
\big\langle j_n\big\rangle(r)=\frac{\Phi}{\Delta\Omega}\,\frac{1}{r^{\,n-1}} .
\label{corollary:info-gas:eq:cone}
\end{equation}
With the Linear Channel Closure \eqref{axioms:linear-closure:eq},
\begin{equation}
\big|\mathbf E_I(r)\big|
=\frac{1}{\kappa}\,\big\langle j_n\big\rangle(r)
=\frac{1}{\kappa}\,\frac{\Phi}{A_n(r)}
\ \propto\ r^{-(n-1)} .
\label{corollary:info-gas:eq:sphere}
\end{equation}

\paragraph{Directional filtering.}
For apertures or one‑sided diaphragms, the inward component is
\[
\Phi_-(r):=\int_{\Sigma(r)}\!\big(\mathbf j\!\cdot \mathbf n\big)_-\,\mathrm dA .
\]
Equations \eqref{corollary:info-gas:eq:tube}–\eqref{corollary:info-gas:eq:sphere} are valid under the substitution \(\Phi\mapsto\Phi_-\).

\paragraph{Window counts \& statistics.}
On a boundary patch \(A\) over \([t,t+\Delta t]\),
\[
N_A(\Delta t):=\int_t^{t+\Delta t}\!\!\int_A j_n\,\mathrm dA\,\mathrm dt,\quad
I_A:=\frac{N_A(\Delta t)}{\Delta t},\quad
\bar j_n:=\Delta t^{-1}\!\int_t^{t+\Delta t} j_n\,\mathrm dt .
\]
The Fano factor \(F_A:=\mathrm{Var}[N_A]/\mathbb E[N_A]\) and an effective window \(\Delta t_{\!\mathrm{eff}}\le\Delta t\) (from the count autocorrelation) summarize fluctuations.
